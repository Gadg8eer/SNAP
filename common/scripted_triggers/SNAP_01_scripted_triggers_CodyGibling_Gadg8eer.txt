Step 1: Load game
### When it doesn't matter what the specific trait is, only that it makes the ruler into that character.
### If the leader has one of these leader traits, this triggers the startup sequence for Gadg8eer as ruler.
MAP_startup_trigger_CodyGibling_Gadg8eer_0 = {
	pre_triggers = {
		has_owner = yes				# whether the leader exists and is alive
	}
	on_game_start = {
		limit = {
			AND = {
				owner = {
					ruler = {
						AND = {
							OR = {
								is_ruler = yes
								is_heir = yes
							OR = {
								has_ruler_trait = leader_trait_MAP_codygibling_gadg8eer
								has_ruler_trait = leader_trait_MAP_codygibling_quick_gadg8eer
								has_ruler_trait = leader_trait_MAP_codygibling_social_gadg8eer
								has_ruler_trait = leader_trait_MAP_codygibling_playing_gadg8eer
							}
						}
					}
				}
			}
		}
	}
}
# Should trigger event; metaverse.avatar.portraits_codygibling.0
# In turn, metaverse.avatar.portraits_codygibling.0 activates global flag; MAP_scripted_effect_codygibling_gadg8eer_start_0
# MAP_scripted_effect_codygibling_gadg8eer_start_0 tells the effects to engage and prepares the next step.

### Step 2: Press begin
MAP_startup_trigger_1 = {
	on_press_begin = {
		limit = {
			has_global flag = MAP_scripted_effect_codygibling_gadg8eer_start_0
		}
		
	}
}
# Should trigger event; metaverse.avatar.portraits_codygibling.1
# In turn, metaverse.avatar.portraits_codygibling.1 activates global flag; MAP_scripted_effect_codygibling_gadg8eer_start_0
# MAP_scripted_effect_codygibling_gadg8eer_start_0 tells the effects to engage and prepares the next step.

MAP_is_gadg8eer = {
	pre_triggers = {
		has_owner = yes				# whether the leader exists and is alive
		is_idle = no				# whether the leader is idle
	}
	if = {
		limit = {
			who = {
				event_target:Gadg8eer = {
					OR = {
						has_ruler_trait = leader_trait_MAP_codygibling_gadg8eer
						has_ruler_trait = leader_trait_MAP_codygibling_quick_gadg8eer
						has_ruler_trait = leader_trait_MAP_codygibling_social_gadg8eer
						has_ruler_trait = leader_trait_MAP_codygibling_playing_gadg8eer
					}
				}
			}
		}
	}
}

### If he turns 86, that's it. He's dead. If he gets fired, he also dies. If he dies, the trigger fires.
MAP_is_deadbutessential_MAP_codygibling_gadg8eer = {
	pre_triggers = {
		has_owner = yes				# whether the leader exists and is alive
	}
	trigger = {
		AND = {
			MAP_is_gadg8eer = true
			who = {
				event_target:Gadg8eer = {
					OR = {
						AND = {
							has_leader_flag = leader_is_MAP_CodyGibling_Gadg8eer
							leader_age = 86
							is_immortal = no
						}
						AND = {
							is_immortal = no
							on_leader_death
							NOT = {
								leader_age > 85
							}
						}
						on_leader_fired
					}
				}
			}
		}
	}
}

### If he would have died of old age, but is immortal without eternal youth, this trigger fires.
MAP_is_deadbutimmortal_MAP_codygibling_gadg8eer = {
	pre_triggers = {
		has_owner = yes				# whether the leader exists and is alive
		is_idle = no				# whether the leader is idle
	}
	trigger = {
		AND = {
			MAP_is_gadg8eer = true
			who = {
				event_target:Gadg8eer = {
					AND = {
						NOR = {
							is_oneiroid_westerner = yes
							is_vidjateer = yes
						}
						is_immortal = yes
						leader_age > 85
					}
				}
			}
		}
	}
}
